//cblastest.c
#include <stdio.h>
#include <stdlib.h>
#include <cblas.h>  // 包含OpenBLAS的头文件

// 定义矩阵的大小
#define MATRIX_SIZE 1000

int main() {
    double *matrixA, *matrixB, *resultMatrix;
    int i, j;

    // 分配内存并初始化矩阵A、矩阵B和结果矩阵
    matrixA = (double *)malloc(MATRIX_SIZE * MATRIX_SIZE * sizeof(double));
    matrixB = (double *)malloc(MATRIX_SIZE * MATRIX_SIZE * sizeof(double));
    resultMatrix = (double *)malloc(MATRIX_SIZE * MATRIX_SIZE * sizeof(double));

    for (i = 0; i < MATRIX_SIZE; i++) {
        for (j = 0; j < MATRIX_SIZE; j++) {
            matrixA[i * MATRIX_SIZE + j] = (double)rand() / RAND_MAX;
            matrixB[i * MATRIX_SIZE + j] = (double)rand() / RAND_MAX;
        }
    }

    // 使用OpenBLAS库中的cblas_dgemm函数执行矩阵相乘
    cblas_dgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, MATRIX_SIZE, MATRIX_SIZE, MATRIX_SIZE, 1.0, matrixA, MATRIX_SIZE, matrixB, MATRIX_SIZE, 0.0, resultMatrix, MATRIX_SIZE);

    // 输出结果矩阵的一部分（可选）
    printf("Result Matrix (first 5x5):\n");
    for (i = 0; i < 5; i++) {
        for (j = 0; j < 5; j++) {
            printf("%f\t", resultMatrix[i * MATRIX_SIZE + j]);
        }
        printf("\n");
    }

    // 释放分配的内存
    free(matrixA);
    free(matrixB);
    free(resultMatrix);

    return 0;
}
