1.股票多次购买获取最大利润

开始便是动态规划的题目，股票问题

给你一个整数数组 prices ，其中 prices[i] 表示某支股票第 i 天的价格。

在每一天，你可以决定是否购买和/或出售股票。你在任何时候 最多 只能持有 一股 股票。你也可以先购买，然后在 同一天 出售。

返回 你能获得的 最大 利润 。

可以多次购买，之间使用贪心解决过了，现在使用动态规划来解决。

1.首先思考dp数组的含义，数组为2维，因为要考虑是否持有股票两种情况。

dp表示是否持有股票的现金。

dp[i][0]表示持有当前股票的现金数量。

dp[i][1]表示不持有当前股票的现金。

2.考虑状态转移

dp[0][]表示第一天，如果持有现金就是-prices[0], 不持有就是0，数组初始值定义了。

dp[i][0]表示要持有股票，考虑前一天持有那就是dp[i-1][1]，前一天未持有就要买入dp[i-1][1]-prices[i]

dp[i][1]表示不持有股票，考虑前一天持有就是dp[i-1][1]+prices[i], 前一天未持有便保持原态dp[i-1][1]

1
2

dp[i][0] = max(dp[i-1][0], dp[i-1][1] - prices[i]);
dp[i][1] = max(dp[i-1][1], dp[i-1][0] + prices[i]);

2. 最大递增子序列长度

给你一个整数数组 nums ，找到其中最长严格递增子序列的长度。

子序列 是由数组派生而来的序列，删除（或不删除）数组中的元素而不改变其余元素的顺序。例如，[3,6,2,7] 是数组 [0,3,1,6,2,2,7] 的子序列。

动态规划，

1.确定dp的含义，一维dp。dp[i]表示以当前为末尾的子序列的最大值。

2.状态转移。那么dp[i]该怎么确定呢，已知dp[i]的求解就是遍历之前所有的点，如果存在dp[i] > dp[j]（即递增关系），那么可以得到dp[j] + 1, 结果取遍历最大值。

vector<vector<int>> fixdis(vector<vector<int>> &mat){
	int m = mat.size(), n = mat[0].size();
	//需要一个计算距离的矩阵
	vector<vector<int>> dis(m, vector<int>(n));//全部初始化为0
	//是否访问过，1为已被访问
	vector<vector<int>> visited(m, vector<int>(n));
	queue<pair<int, int>> q;//一个队列，用于处理0
	
	//第一步，需要把所有的0全部入一个队列，依次处理
	for(int i = 0; i < m; i++){
		for(int j = 0; j < n; j++){
			if(mat[i][j] == 0){
				q.push(make_pair(i, j));
				visited[i][j] = 1;
			}
		}
	}
	
	//第二部，循环处理
	while(!q.empty()){
		auto [i, j] = q.front();//从队列前取出一对，格式就是这样
		q.pop();
		//这里开始处理当前节点周围距离为1的节点，矩阵是只有4个方向，考虑可以使用一个数组来表示。
		for(int d = 0; d < 4; d++){
			int ci = i + dire[d][0];
			int cj = j + dire[d][1];
			//判断是否处理过，或者有无越界
			if(ci >= 0 && ci < m && cj >= 0 && cj < n && visited[ci][cj] != 1){
				dist[ci][cj] = dist[i][j] + 1;
				visited[ci][cj] = 1;

			}

void dfs(String s, int index){
    int len = s.length();
    if(track.size() == 4 && index != len){//不合条件
        return;
    }
    if(track.size() == 4 && index == len){
        res.add(track.get(0) + "." + track.get(1) + "." + track.get(2) + "." + track.get(3));//拼接字符串
    }
    for(int i = index, j = i+1; j < len+1 && j-i <= 3; j++){//dfs递归的是深度，每往下就track size加1， 里面是这一层的回溯逻辑
        if(validata(s.subString(i, j))){
            track.add(s.subString(i, j));
            dfs(s, j);
            track.remove(track.size()-1);
        }
    }
}

uint32_t reverseBits(uint32_t n){
	uint32_t result = 0;
	int i = 0;
	while(i--){
		result <<= 1;
		result += n&1;
		n >>=1;
	}
	return result;
}

public ListNode rL(ListNode head){
	if(head == null) return null;
    ListNode dum = new ListNode(0);//创建一个前置节点，这个节点是不动的
    ListNode pre = dum;//pre节点，意义可知
    dum.next = head;
    ListNode next = head.next;
    while(next != null){//这个题是判空的模式，有的题目可能是翻转的次数的循环
        head.next = next.next;
        next.next = pre.next;
        pre.next = next;
        next = head.next;//next一直是head之后的一个
        
    }
    return dum.next;
}

public ArrayList<ArrayList<Integer>> levelOrder (TreeNode root) {
        // write code here
        ArrayList<ArrayList<Integer>> alists = new ArrayList<ArrayList<Integer>>();
        ArrayList<Integer> list = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<TreeNode>();
        if(root == null){
            return alists;
        }
        q.offer(root);
        TreeNode cur = root;
        TreeNode last = root;//记住一行最后一个节点,这个需要临时保存一段时间。
        TreeNode nlast = root;
        while(!q.isEmpty()){
            cur = q.poll();
            if(cur.left != null){
                q.offer(cur.left);
                nlast = cur.left;
            }
            if(cur.right != null){
                q.offer(cur.right);
                nlast = cur.right;
            }
            list.add(cur.val);
            if(cur == last){
                alists.add(list);
                list = new ArrayList<>();
                last = nlast;
            }
        }
        
        return alists;
    }

        }
    }
    while(i <= mid) t[k++] = nums[i++];
    while(j <= end) t[k++] = nums[j++];
    System.arraycopy(t, 0, nums, begin, index);
}

public int nqueen(int n){

int res = 0;
void backTracking(int[] queens, int row){
    if(row == n){
        res++;
        return;
	}
    for(int i = 0; i < queens.length; i++){
        //这个循环表示列
        if(check(queens, row, i)){
            queens[row] = i;
            backTracking(queens, row+1)；
            //不用往会置位是因为一维数组会刷新
        }
	}
}

boolean check(int[] queens, int row, int col){
    for(int i = 0; i < row; i++){//判断前row-1行，因为后面还没开始填充
        if(queens[row] == col || Math.abs(row - i) == Math.abs(queens[i] - col)){
            return false;
        }
    }
    return true;
}

    PriorityQueue<Integer> maxHeap = new PriorityQueue<Integer>(11, new Comparator<Integer>(){

        }
    });//构建大顶堆，传入一个comparetor
	public void insert(Integer num){
        if(count % 2 == 0){//说明此时两个堆已经平衡，大小分明，如果传入一个元素，我们规定放到小堆里，直接放进去肯定不行，因为可能这个数可能比大堆的数小，所以置换出大堆里的元素，放到小堆就可以
            maxHeap.add(num);
            minHeap.add(maxHeap.poll());
        }else{//小堆多一个元素，需要判断，如果数据大于小堆，说明这个数据应该在小堆，置换到出来放到大堆里即可
            if(num > minHeap.peak()){
                minHeap.add(num);
                maxHeap.add(minHeap.poll());
			}else{//小于小堆，说明这个数应该在大堆，直接入大堆即可
                maxHeap.add(num);
            }
        }
        count ++;
    }

            return ((double)minHeap.peak() + maxHeap.peak()) / 2;

}

public int solve (String s) {
        // write code here
        Map<Character, Integer> map = new HashMap<>();
        map.put('+', 1);
        map.put('-', 1);
        map.put('*', 2);
        Deque<Integer> nums = new ArrayDeque<>();
        Deque<Character> opts = new ArrayDeque<>();//两个栈，数字栈和符号栈
        s.replaceAll(" ", "");//去空格
        char[] chs = s.toCharArray();//转为字符数组
        int n = chs.length;

        for(int i = 0; i < n; i++){
            char c = chs[i];
            if(isNumber(c)){//情况1，如果是数字，直接压入数字栈
                int num = 0;
                int j = i;
                while(j < n && isNumber(chs[j])){//读连续的数字
                    num = 10 * num + chs[j++] - '0';
                }
                nums.push(num);
                i = j - 1;
            }else if(c == '('){//情况2：遇到左括号直接入栈
                opts.push(c);
            }else if(c == ')'){//情况3：遇到右括号，计算左括号之前的
                while(opts.peek() != '('){
                    cal(nums, opts);
                }
                opts.pop();
            }else{//情况4:字符，判断优先级，栈顶的字符优先级大等于当前字符，计算，直到栈空或者遇到'(';字符入栈
                while(!opts.isEmpty() && opts.peek() != '(' && map.get(opts.peek()) >= map.get(c)){
                    cal(nums, opts);
                }
                opts.push(c);
            }
        }
        while(!opts.isEmpty()){
            cal(nums, opts);
        }

        return nums.pop();
    }

private boolean isNumber(Character c){
        return Character.isDigit(c);
    }
    private void cal(Deque<Integer> nums, Deque<Character> opts){
        int num2 = nums.pop();
        int num1 = nums.pop();
        char opt = opts.pop();
        if(opt == '+'){
            nums.push(num1+num2);
        }else if(opt == '-'){
            nums.push(num1-num2);
        }else if(opt == '*'){
            nums.push(num1 * num2);
        }
    }

public int minMoney (int[] arr, int aim) {
        // write code here
        if(aim == 0){
            return 0;
        }
        int dp[] = new int[aim+1];
        Arrays.fill(dp, 1, dp.length, aim+1);
        for(int i = 1; i <= aim; i++){
            for(int coin : arr){
                //分不出来或者前面的分配失败
                if(i-coin<0 || dp[i-coin]>aim) continue;
                dp[i] = Math.min(dp[i-coin] + 1, dp[i]);
            }
        }
        return dp[aim] == aim + 1 ? -1 : dp[aim];
    }

//思路，是一个动态规划，整体上考虑，i，j表示字符串从i到j的子串，dp[i][j] = dp[i+1][j-1] + 2
//也就是说这个i需要从大到小，而j从小到大
public int getLongestPalindrome (String A) {
        // write code here
        //二维数组表示，每一行从【i,j】表示下标i到下标j
        int n = A.length();
        int[][] dp = new int[n][n];
        int res = 1;//
        for(int i = n - 1; i >= 0; i--){
            dp[i][i] = 1;
            for(int j = i+1; j < n; j++){
                //i,j字符相同，看i+1和j-1，如果dp[i+1][j-1]不为0，加2，如果i+1 >= j-1了，也+2
                if(A.charAt(i) == A.charAt(j) && (dp[i+1][j-1] != 0 || i+1 >= j-1)){
                    dp[i][j] = dp[i+1][j-1] +2;
                }else{
                    dp[i][j] = 0;
                }
                res = Math.max(res, dp[i][j]);
            }
        }
        return res;
    }
