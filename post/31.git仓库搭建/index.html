# 创建git仓库目录
mkdir /home/git
# 设置权限
chown git:git /home/git
chmod 755 /home/git

cd /home/git
# 切换到git用户
su git

在git用户目录下面创建.ssh目录
mkdir .ssh

cd .ssh
# 
touch authorized_keys

# 把个人pc的ssh公钥复制到上面的文件里
# 然后设置权限
chmod 600 ~/.ssh/authorized_keys

三、创建git仓库

在/home/git/目录下创建自己的git仓库

1

git init --bare demo.git

设置仓库权限

1
2

sudo chown -R git:git /home/git/demo.git
sudo chmod -R 755 /home/git/demo.git

在个人pc的git bash里可以clone仓库

1

git clone git@xx.xx.xx:/home/git/demo.git

后续可以对仓库代码进行管理。
