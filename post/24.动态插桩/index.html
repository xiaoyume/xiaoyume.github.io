#include<stdio.h>
#include<stdlib.h>
#include<stdint.h>
#include <math.h>
#include <time.h>
#include"tools.h"
//#include "hthread_host.h"
//#include "mt_hybrid.h"
#include"mtblas.h"
#include<string.h>
#include<sys/time.h>
#include <omp.h>

//lammps里使用的接口
void dgemm_(char *ta, 
char *tb,
int* M,
int* N,
int* K,
double* alpha,
const double* A,
int* lda,
const double* B,
int* ldb,
double* beta,
double* C,
int* ldc){
	int cluster_id = 0;
	mt_blas_init(cluster_id);
	int M_t = *M, N_t = *N, K_t = *K;
	int64_t lda_t = (int64_t)*lda;
	int64_t ldb_t = (int64_t)*ldb;
	int64_t ldc_t = (int64_t)*ldc;
	float alpha_t = (float)*alpha;
	float beta_t = (float)*beta;
	double *A_t, *B_t, *C_t;
	int q = 0;
	A_t = hthread_malloc(q, 8*M_t*lda_t, HT_MEM_RO);
	B_t = hthread_malloc(q, 8*K_t*ldb_t, HT_MEM_RO);
	C_t = hthread_malloc(q, 8*M_t*ldc_t, HT_MEM_RW);
	memcpy(A_t, A, 8*M_t*K_t);
	memcpy(B_t, B, 8*K_t*N_t);
	printf("DSP begin computing!\n");
	mt_hthread_dgemm(M_t, N_t, K_t, alpha_t, A_t, lda_t, B_t, ldb_t, beta_t, C_t, ldc_t , cluster_id);
	memcpy(C, C_t, 8*M_t*N_t);
	printf("Success!\n");
	hthread_free(A_t);
	hthread_free(B_t);
	hthread_free(C_t);
	hthread_dev_close(cluster_id);
}
