ServerSocket ss = new ServerSocket(8888);//指定端口号

Socket s = ss.accept();//等待客户端连接
system.out.println("客户端：" + s.getInetAddress().getLocalHost() + "已连接到服务器");//打印连接信息

//创建字符缓冲流
//InputStreamReader读入字节流解析为字符流
BufferedReader br = new BufferedReader(new InputStreamReader(s.getInputStream()));
String msg = br.readLine();//读出一行客户端字节流
System.out.println("来自客户端的信息：[" + msg + "]");

BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(s.getOutputStream()));
bw.write("这是服务端发送的信息\n" );
bw.flush();
